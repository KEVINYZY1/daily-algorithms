```js
function resolve(n) {
  var map = [
    [0, "M", "MM", "MMM"],
    [0, "C", "CC", "CCC", "CD", "D", "DC", "DCC", "DCCC", "CM"],
    [0, "X", "XX", "XXX", "XL", "L", "LXX", "LXX", "LXXX", "XC"],
    [0, "I", "II", "III", "IV", "V", "VI", "VII", "VIII", "IX"]
  ];
  return map[0][~~(n / 1000)] + 
         map[1][~~(n % 1000 / 100)] + 
         map[2][~~(n % 100 / 10)] + 
         map[3][n % 10];
}

console.assert(resolve(1437) === 'MCDXXXVII', resolve(1437));
```
